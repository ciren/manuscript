\makeatletter
\long\def\@makecaption#1#2{%
        \vskip\abovecaptionskip
        \sbox\@tempboxa{{\small{\bf #1:} #2}}%
        \ifdim \wd\@tempboxa >\hsize
                {\small{\bf #1:} #2\par}
        \else
                \hbox to\hsize{\hfil\box\@tempboxa\hfil}%
        \fi
        \vskip\belowcaptionskip
}
